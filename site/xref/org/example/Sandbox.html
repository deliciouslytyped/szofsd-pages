1   package org.example;

3   import javafx.application.Application;
4   import javafx.application.ConditionalFeature;
5   import javafx.application.Platform;
6   import javafx.scene.Group;
7   import javafx.scene.PerspectiveCamera;
8   import javafx.scene.Scene;
9   import javafx.scene.shape.Box;
10  import javafx.scene.shape.CullFace;
11  import javafx.scene.transform.Rotate;
12  import javafx.stage.Stage;
13  
14  // Edit test
15  
16  public class Sandbox extends Application {
17  
18      public static void main(String[] args) {
19          launch(args);
20      }
21  
22      public void start(Stage primaryStage) {
23  
24          boolean is3DSupported = Platform.isSupported(ConditionalFeature.SCENE3D);
25          if(!is3DSupported) {
26              System.out.println("Sorry, 3D is not supported in JavaFX on this platform.");
27              return;
28          }
29  
30          Box box = new Box(100,100,100);
31          box.setCullFace(CullFace.NONE);
32          box.setTranslateX(250);
33          box.setTranslateY(100);
34          box.setTranslateZ(400);
35  
36          boolean fixedEyeAtCameraZero = false;
37          PerspectiveCamera camera = new PerspectiveCamera(fixedEyeAtCameraZero);
38          camera.setTranslateX(150);
39          camera.setTranslateY(-100);
40          camera.setTranslateZ(250);
41  
42          Group root = new Group(box);
43          root.setRotationAxis(Rotate.X_AXIS);
44          root.setRotate(30);
45  
46          Scene scene = new Scene(root, 500, 300, true);
47          scene.setCamera(camera);
48          primaryStage.setScene(scene);
49          primaryStage.setTitle("3D Example");
50  
51          primaryStage.show();
52      }
53  }
